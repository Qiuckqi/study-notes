1. 查看文件/目录
	> ls 		// 列出目录文件信息
	> ls -l 	// 查看详细信息
	> ls -a		//查看全部文件 包括隐藏文件
	> ls -al	// 查看全部文件的详细信息

2. 目录切换
	> cd dirname 	//切换目录
	> cd ..			//切换到上机目录
	> cd ~			//直接切换到当前用户的家目录
	
3. 查看完整路径
	> pwd		
	
4. 用户切换
	> su - 或 su - root 	//向root用户切换
	> exit 			//退会上一级用户
	> su 用户名		//普通用户切换
	多次使用su指令会当值用户的叠加
	su和exit指令最好配对使用
	
5. 查看当前用户
	> whoami
	
6. 在图形界面和命令界面切换
	root用户才能切换
	># init 3	//进入命令界面
	># init 5	//进入图形界面
	
7. 查看指令对应的执行文件程序路径
	> which 指令
